#include<iostream>
using namespace std;
const int N = 100010,M = 1000010;
char p[N],s[M];
int n,m;
int ne[N];//NEXT数组
int main()
{
    cin >> n >> p+1 >> m >> s+1;
    //求next数组过程
    for(int i=2,j=0;i<=n;i++)
    {
        while(j && p[i]!=p[j+1])  j = ne[j];
        if(p[i] == p[j+1])  j++;
        ne[i] = j;
    }
    //KMP匹配过程
    for(int i=1,j=0;i<=m;i++)
    {
        while(j && s[i]!=p[j+1])    j = ne[j];//不匹配
        if(s[i] == p[j+1])  j++;//匹配
        if(j==n){
            printf("%d ",i-n);//匹配成功的初始下标
            j = ne[j];
        }
    }
    return 0;
}
