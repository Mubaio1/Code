#include<bits/stdc++.h>
using namespace std;
const int N = 1010;
int f[N][N];
int v[N],w[N];
int n,m;

int main()
{
    int res = 0;
    cin >> n >> m;
    for(int i=1;i<=n;i++)
    {
        cin >> v[i] >> w[i];
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++){
            f[i][j] = f[i-1][j];//记录不包含i的时候背包值并处理j<v[i]时的情况
            if(j >= v[i]) f[i][j] = max(f[i-1][j],f[i-1][j-v[i]]+w[i]);//比较不包含i和包含i的值
        }
    }
    cout << f[n][m] << endl;
    return 0;
}
