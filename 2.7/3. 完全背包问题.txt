#include<bits/stdc++.h>
using namespace std;
const int N = 1010;
int f[N];
int v[N],w[N];
int n,m;

int main()
{
    int res = 0;
    cin >> n >> m;
    for(int i=1;i<=n;i++)
    {
        cin >> v[i] >> w[i];
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=v[i];j<=m;j++){//可以重复使用某一个
            f[j] = max(f[j],f[j-v[i]]+w[i]);
        }
    }
    cout << f[m] << endl;
    return 0;
}
