class Solution {
public:
    int minRemoval(vector<int>& nums, int k) {
        int n = nums.size();
        int cnt = 0;
        if(n == 1) return cnt;
        sort(nums.begin(),nums.end());
        int left=0;
        for(int i=0;i<n;i++){
            while(1LL*nums[left]*k < nums[i]){//1LL是longlong整形的1，类型提升
                left++;
            }
            cnt = max(cnt,i-left+1);//最大的平衡数组长度
        }
        return n-cnt;
    }
};
