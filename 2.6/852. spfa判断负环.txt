#include<bits/stdc++.h>
using namespace std;

const int N = 2010;
const int M = 10010;
int h[N],ne[M],w[M],e[M];
int dist[N];
int cnt[N];
bool st[N];
int n,m;
int idx;
void add(int a,int b,int c)
{
    e[idx] = b;
    w[idx] = c;
    ne[idx] = h[a];
    h[a] = idx++;
}

bool spfa()
{
    //memset(dist,2e7,sizeof dist);//只有当出现负权值的时候更新  设为最大会超时 设为nm就不会超时或者不初始化
    memset(cnt,0,sizeof cnt);
    queue<int> q;
    
    for(int i=1;i<=n;i++){
        st[i] = true;
        q.push(i);
    } 
    
    while(q.size())
    {
        auto t = q.front();
        q.pop();
        st[t] = false;
        for(int i=h[t];i!=-1;i=ne[i])
        {
            int j = e[i];
            if(dist[j] > dist[t]+w[i])
            {
                dist[j] = dist[t]+w[i];
                cnt[j] = cnt[t]+1;
                if(cnt[j] >= n) return true;
                if(!st[j]){
                    q.push(j);
                    st[j] = true;
                } 
            }
        }
    }
    return false;
}

int main()
{
    memset(h,-1,sizeof h);
    cin >> n >> m;
    while(m--){
        int a,b,c;
        cin >> a >> b >> c;
        add(a,b,c);
    }
    if(spfa()) cout << "Yes" << endl;
    else cout << "No" << endl;
    return 0;
}
