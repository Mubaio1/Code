#include<bits/stdc++.h>
using namespace std;
const int N = 100010;
int b[N],a[N];
int ra,rb;
int main()
{
    int n,m,x;
    cin >> n >> m >> x;
    for(int i=0;i<n;i++) cin >> a[i];
    for(int i=0;i<m;i++) cin >> b[i];
    ra = 0;
    rb = m-1;
    while(ra<n && rb>=0){
        int sum = a[ra]+b[rb];
        if(sum == x){
            cout << ra << " " << rb << endl;
            break;
        }
        else if(sum < x) ra++;
        else rb--;
    }
    return 0;
}
