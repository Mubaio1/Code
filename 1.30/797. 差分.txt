#include<bits/stdc++.h>
using namespace std;
const int N = 100010;
int a[N];
int b[N];

void insert(int l,int r,int c)
{
    b[l]+=c;
    b[r+1]-=c;
}

int main()
{
    int n,m;
    cin >> n >> m;
    
    for(int i=1;i<=n;i++){
        scanf("%d",&a[i]);
    }
    for(int i=1;i<=n;i++){
        insert(i,i,a[i]);//差分初始化  对于每一个b都是后一个a减前一个a的值
    }
    while(m--){
        int l,r,c;
        cin >> l >> r >> c;
        insert(l,r,c);
    }
    for(int i=1;i<=n;i++){
        b[i] += b[i-1];
    }
    for(int i=1;i<=n;i++){
        cout << b[i] << " ";
    }
    return 0;
}
