#include<bits/stdc++.h>
using namespace std;

int a[1010][1010];
int b[1010][1010];


//
void insert(int x1,int y1,int x2,int y2,int c)//更新初始为全0的差分数组
{
    b[x1][y1]+=c;
    b[x1][y2+1]-=c;
    b[x2+1][y1]-=c;
    b[x2+1][y2+1]+=c;
}

int main()
{
    int n,m,q;
    cin >> n >> m >> q;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            cin >> a[i][j];
        }
    }
    for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                insert(i,j,i,j,a[i][j]);//对于每一个小矩形进行更新，视为对初始为全0的差分矩阵进行插入构造
            }
    }
    while(q--){
        int x1,y1,x2,y2,c;
        cin >> x1 >> y1 >> x2 >> y2 >> c;
        insert(x1,y1,x2,y2,c);
    }
        
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            b[i][j] = b[i][j]-b[i-1][j-1]+b[i-1][j]+b[i][j-1];//通过差分数组求前缀和
            cout << b[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
