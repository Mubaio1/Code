#include<bits/stdc++.h>
using namespace std;

const int N = 100003;

int h[N],e[N],ne[N],idx;

void insert(int x)
{
    int t = (x%N+N)%N;//保证为正整数
    e[idx] = x;
    ne[idx] = h[t];
    h[t] = idx++;//头插法指向该链的第一个位置
}

bool find(int x)
{
    int t = (x%N+N)%N;//保证为正整数
    for(int i=h[t];i!=-1;i=ne[i]){
        if(e[i] == x) 
        {
            cout << "Yes" << endl;
            return true;
        }
    }
    cout << "No" << endl;
    return false;
}

int main()
{
    int n;
    cin >> n;
    memset(h,-1,sizeof(h));
    while(n--){
        char op;
        int x;
        cin >> op >> x;
        if(op == 'I'){
            insert(x);
        }
        else{//op == 'Q'
            find(x);
        }
    }
    return 0;
}
