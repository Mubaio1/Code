#include<bits/stdc++.h>
using namespace std;
const int N = 1010;

int f[N][N];
char a[N],b[N];

int n,m;

int main()
{
    cin >> n >> a+1 >> m >> b+1;
    
    //赋初值
    for(int i=0;i<=m;i++) f[0][i] = i;
    for(int i=0;i<=n;i++) f[i][0] = i;
    
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            //对四种操作分别进行讨论,首先讨论增和删  其次判断最后一位是否相等决定改或者不改
            f[i][j] = min(f[i-1][j],f[i][j-1])+1;
            
            if(a[i] == b[j]) f[i][j] = min(f[i][j],f[i-1][j-1]);
            else f[i][j] = min(f[i][j],f[i-1][j-1]+1);
        }
    }
    cout << f[n][m] << endl;
    return 0;
}#include<bits/stdc++.h>
