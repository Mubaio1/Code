#include<bits/stdc++.h>
using namespace std;
const int N = 15;
const int M = 1010;
int f[N][N];
char str[M][N];

int n,m;

int edit_distance(char a[],char b[])
{
    //由于调用时传入的参数是str[i]和s,所以这里的a和b是从0开始的，我们计算长度要从a+1和b+1开始
    int la = strlen(a+1);
    int lb = strlen(b+1);
    //赋初值
    for(int i=0;i<=lb;i++) f[0][i] = i;
    for(int i=0;i<=la;i++) f[i][0] = i;
    
    for(int i=1;i<=la;i++)
    {
        for(int j=1;j<=lb;j++)
        {
            //对四种操作分别进行讨论,首先讨论增和删  其次判断最后一位是否相等决定改或者不改
            f[i][j] = min(f[i-1][j],f[i][j-1])+1;
            f[i][j] = min(f[i][j],(f[i-1][j-1]+(a[i]!=b[j])));
        }
    }
    return f[la][lb];
}

int main()
{
    cin >> n >> m;
    for(int i=0;i<n;i++)
    {
        cin >> str[i]+1;
    }
    

    while(m--)
    {
        char s[N];
        int limit;
        cin >> s+1 >> limit;
        int res = 0;
        for(int i=0;i<n;i++)
        {
            if(edit_distance(str[i],s) <= limit) res++;
            // cout << edit_distance(str[i],s) << endl;
        }
        cout << res << endl;
    }
    return 0;
}
