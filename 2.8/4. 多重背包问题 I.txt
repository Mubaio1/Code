#include<bits/stdc++.h>
using namespace std;
const int N = 1010;
int f[N][N];
int v[N],w[N],s[N];
int n,m;

int main()
{
    int res = 0;
    cin >> n >> m;
    for(int i=1;i<=n;i++)
    {
        cin >> v[i] >> w[i] >> s[i];
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<=m;j++){
            for(int k=0;k<=s[i] && k*v[i]<=j;k++){//k必须小于每组的个数，以及总体积不能大于j
                f[i][j] = max(f[i][j],f[i-1][j-k*v[i]]+k*w[i]);
            }
        }
    }
    cout << f[n][m] << endl;
    return 0;
}
