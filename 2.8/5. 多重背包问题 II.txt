#include<bits/stdc++.h>
using namespace std;
const int N = 1010 * 12;
const int M = 2010;
int f[M];//体积的状态数
int v[N],w[N];//这里的大小为物品种类数和每组最多打包个数的乘积
int n,m;

//多重背包二进制打包转为0-1背包问题
int main()
{
    cin >> n >> m;
    int cnt=0;
    for(int i=1;i<=n;i++)
    {
        int a,b,s;
        cin >> a >> b >> s;
        int k = 1;
        while(k<=s)
        {
            cnt++;
            v[cnt] = a * k;
            w[cnt] = b * k;
            s -= k;
            k = k*2;
        }
        if(s>0)
        {
            cnt++;
            v[cnt] = a * s;
            w[cnt] = b * s;
        }
    }
    
    //
    n = cnt;//将n个状态更新为cnt个状态
    //
    for(int i=1;i<=n;i++){
        for(int j=m;j>=v[i];j--)
        {
            f[j] = max(f[j],f[j-v[i]]+w[i]);
        }
    }
    cout << f[m] << endl;
    return 0;
}
